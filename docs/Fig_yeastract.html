# reading in matrices
readInRegulatoryMatrix <- function(.file_name) {
  matdf <- read_delim(.file_name,
                      delim = ";")
  mat <- matdf[, -1] |> as.matrix()
  colnames(mat) <- left_join(tibble(common = colnames(mat)),
                             yeastract_lookup, by = "common") |> 
    dplyr::select(systematic) |> pull()
  rownames(mat) <- left_join(tibble(common_protein = unlist(matdf[,1])),
                             yeastract_lookup, by = "common_protein") |> 
    dplyr::select(systematic) |> pull()
  return(mat)
}

Visualizing environment-specific networks

Given the adjacency matrix, outputs the connected portion of the graph in igraph format, good for plotting:

getConnectedGraph <- function(.adj) {
  colnames(.adj) <- left_join(tibble(systematic = colnames(.adj)),
                             yeastract_lookup, by = "systematic") |> 
    dplyr::select(common) |> pull()
  rownames(.adj) <- left_join(tibble(systematic = rownames(.adj)),
                             yeastract_lookup, by = "systematic") |> 
    dplyr::select(common) |> pull()
  regnames <- rownames(.adj)[rowSums(.adj) > 0]
  g <- graph_from_adjacency_matrix(.adj)
  sub_g <- subgraph(g, which(igraph::degree(g) > 0))
  V(sub_g)$isHub <- names(V(sub_g)) %in% regnames
  V(sub_g)$vertex_size <- if_else(V(sub_g)$isHub, true = 7, false = 1)
  V(sub_g)$color <- if_else(V(sub_g)$isHub, true = "steelblue", false = "orange")
  V(sub_g)$vertex_label <- if_else(V(sub_g)$isHub, true = names(V(sub_g)), false = "")
  return(sub_g)
}

  4. Spar-unique plastic
  5. plasticity reversal (YPD and env-specific connections in each)
# helper for plotting function
getPlasticityGroup <- function(.cer, .par) {
  if (.cer == 0 & .par == 0) {
    return("conserved static")
  }
  if (.cer == .par) {
    return("conserved plastic")
  }
  if (.cer == 0) {
    return("Spar-unique")
  }
  if (.par == 0) {
    return("Scer-unique")
  }
  else {
    return("reversal")
    }
}

plotHomeAndAltEnvironments <- function(.tf_common, 
                                       .reg_idxs = NULL,
                                       .e, .e2, .regulator_plotlims = NULL,
                                       .regulon_plotlims = NULL,
                                       .include_hybrid = TRUE,
                                       .regulator_normalization = "log2",
                                       .regulon_normalization = "scale") {
  e_name <- LongExperimentNames[ExperimentNames == .e]
  e2_name <- LongExperimentNames[ExperimentNames == .e2]
  ### Home environment
  # regulator
  reg_idx <- yeastract_lookup |> filter(common == .tf_common) |>
    dplyr::select(all_of("systematic")) |> pull()
  p_regulator <- annotate_figure(plotGenes(.gene_idxs = reg_idx, .quartet = .include_hybrid,
                                           .normalization = .regulator_normalization,
                                           .plotlims = .regulator_plotlims,
                                           .experiment_name = .e), top = .tf_common) 
  # regulon
  if (is.null(.reg_idxs)) {
    regulon_idxs <- filter(regdf, environment == .e & regulator == reg_idx) |> 
    dplyr::select(target) |> pull() |> 
    intersect(y = finaldf[finaldf$experiment == .e,]$gene_name)
  } else {
    regulon_idxs <- intersect(x = .reg_idxs,
                              y = finaldf[finaldf$experiment == .e,]$gene_name)
  }
  p_regulon <- annotate_figure(plotGenes(.gene_idxs = regulon_idxs, 
                                         .normalization = .regulon_normalization,
                                         .plotlims = .regulon_plotlims,
                                         .experiment_name = .e,
                                         .quartet = .include_hybrid),
                               top = paste0(.tf_common, " regulon (", length(regulon_idxs), " genes)"))
  p_home <- annotate_figure(ggarrange(p_regulator, p_regulon, nrow = 2, ncol = 1),
                            top = e_name)
  ### Alt environment
  # regulator
  p_regulator <- annotate_figure(plotGenes(.gene_idxs = reg_idx, .quartet = .include_hybrid,
                                           .normalization = .regulator_normalization,
                                           .plotlims = .regulator_plotlims,
                                           .experiment_name = .e2), top = .tf_common) 
  # regulon
  p_regulon <- annotate_figure(plotGenes(.gene_idxs = regulon_idxs, 
                                         .normalization = .regulon_normalization,
                                         .plotlims = .regulon_plotlims,
                                         .experiment_name = .e2,
                                         .quartet = .include_hybrid),
                               top = paste0(.tf_common, " regulon (", length(regulon_idxs), " genes)"))
  p_divEnv <- annotate_figure(ggarrange(p_regulator, p_regulon, nrow = 2, ncol = 1),
                              top = e2_name)
  return(ggarrange(p_home, p_divEnv, nrow = 1, ncol = 2))
}

Do any TFs Granger-cause a difference in their targets’ expression? (Archived this analysis b/c it was very sensitive to the number of timepoints in each experiment, but keeping the function in case it is useful)

testGrangerCause <- function(.tf, .target, .environment, .allele, .parents_or_hybrid) {
  if (.parents_or_hybrid == "hybrid") {
    cts_mat <- collapsed_allele[[.allele]]
    info_df <- info_allele
  }
  if (.parents_or_hybrid == "parents") {
    cts_mat <- collapsed[[.allele]]
    info_df <- info
  }
  if (!(.tf %in% rownames(cts_mat)) | !(.target %in% rownames(cts_mat))) {
    return(NA)
  }
  if (.tf == .target) { # no self-feedback allowed
    return(NA)
  }
  test_df <- tibble("regulator" = cts_mat[.tf,
                                          info_df$experiment == .environment],
                    "target" = cts_mat[.target,
                                       info_df$experiment == .environment],
                    "timepoint" = info_df$time_point_num[info_df$experiment == .environment]) |> 
    arrange(timepoint)
  if (var(test_df$regulator) == 0 | var(test_df$target) == 0) { # grangertest returns something other than NA for these, we want it to return NA
    return(NA)
  }
  test <- tryCatch(expr = {grangertest(target ~ regulator, order = 2, data = test_df)},
                   error = function(e) {return(NA)})
  if (!is.na(test)[1]) {
    return(as.numeric(test$`Pr(>F)`[2]))
  }
  return(test)
}
